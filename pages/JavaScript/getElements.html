Get Elements

var entity = document.createElement('tagName') - create tagName tag in HTML.

var entity = document.querySelector('.className') - get the first 
element of the class.
var entity = document.querySelectorAll('.className') - get the collection
 of the elements with the current className.

var entity = document.getElementsById('idName') - get the element 
with current ID.
var entity = document.getElementsByClassName('className') - get the 
collection of the elements with the current className;
var entity = document.getElementsByTagName('tagName') - get the 
collection of the elements with the tagName;
